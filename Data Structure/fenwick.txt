class Fenwick
{
public:
	VI tree;

	Fenwick(const int & n) : tree(n + 1) {}

	int sum(int pos)
	{
		++pos;
		int ret = 0;
		while (pos > 0)
		{
			ret += tree[pos];
			pos &= (pos - 1);
		}
		return ret;
	}

	inline void add(int pos, const int & val)
	{
		++pos;
		while (pos < tree.size())
		{
			tree[pos] += val;
			pos += (pos & -pos);
		}
	}
};
